## Code Blocks

Syntax highlighted code with copy button.

```typescript title="story.ts"
interface Story {
  id: string;
  title: string;
  description: string;
  status: 'todo' | 'in-progress' | 'done';
  points: number;
  assignee?: string;
}

function createStory(title: string): Story {
  return {
    id: `US-${Date.now()}`,
    title,
    description: '',
    status: 'todo',
    points: 1,
  };
}
```
